function switchPanel(panelName) { document.querySelectorAll(".nav button").forEach((button) => { button.classList.toggle("active", button.dataset.panel === panelName); }); document.querySelectorAll(".panel").forEach((panel) => { panel.classList.toggle("active", panel.id === `panel-${panelName}`); }); document.getElementById("page-title").textContent = pageMeta[panelName].title; document.getElementById("page-copy").textContent = pageMeta[panelName].copy; } function render() { renderStats(); renderSummary(); renderLocales(); renderTasks(); renderIncidents(); } function renderStats() { const occupied = state.locales.filter((item) => item.status === "ocupado").length; const activeTasks = state.tareas.filter((item) => item.status !== "completada").length; const openIncidents = state.incidentes.filter((item) => item.status !== "resuelto").length; document.getElementById("stat-locales").textContent = String(state.locales.length); document.getElementById("stat-ocupados").textContent = String(occupied); document.getElementById("stat-tareas").textContent = String(activeTasks); document.getElementById("stat-incidentes").textContent = String(openIncidents); document.getElementById("locales-count-badge").textContent = `${state.locales.length} locales`; document.getElementById("tareas-count-badge").textContent = `${state.tareas.length} tareas`; document.getElementById("incidentes-count-badge").textContent = `${state.incidentes.length} incidentes`; } function renderSummary() { if (!state.locales.length && !state.tareas.length && !state.incidentes.length) { dashboardSummary.innerHTML = emptyState( "Todavia no hay informacion cargada", "Empieza por agregar un local, una tarea o un incidente. El dashboard se completa solo a medida que registres datos." ); return; } const cards = [ { title: "Ultimo local registrado", text: state.locales[0] ? `${state.locales[0].number} - ${state.locales[0].name}` : "Aun no hay locales cargados." }, { title: "Ultima tarea registrada", text: state.tareas[0] ? `${state.tareas[0].title} - ${capitalize(state.tareas[0].status)}` : "Aun no hay tareas cargadas." }, { title: "Ultimo incidente registrado", text: state.incidentes[0] ? truncate(state.incidentes[0].description, 70) : "Aun no hay incidentes cargados." } ]; dashboardSummary.innerHTML = cards.map((item) => `

${escapeHtml(item.title)}

${escapeHtml(item.text)}
`).join(""); } function renderLocales() { if (!state.locales.length) { document.getElementById("locales-count-badge").textContent = "0 locales"; localesList.innerHTML = emptyState( "No hay locales registrados", "Agrega el primer local para iniciar la base comercial." ); return; } const filteredLocales = state.locales.filter((item) => { const query = normalize(filters.locales.query); const matchesQuery = !query || normalize([ item.number, item.name, item.category, item.tenant, item.phone ].join(" ")).includes(query); const matchesStatus = filters.locales.status === "todos" || item.status === filters.locales.status; return matchesQuery && matchesStatus; }); document.getElementById("locales-count-badge").textContent = formatVisibleCount(filteredLocales.length, state.locales.length, "locales"); if (!filteredLocales.length) { localesList.innerHTML = emptyState( "No hay coincidencias para este filtro", "Prueba limpiando el buscador o cambiando el estado seleccionado." ); return; } localesList.innerHTML = filteredLocales.map((item) => `

${escapeHtml(item.number)} - ${escapeHtml(item.name)}

${escapeHtml(item.category || "Sin rubro")} ${escapeHtml(item.tenant || "Sin responsable")} ${escapeHtml(item.phone || "Sin telefono")}
${renderBadge(item.status, "locale")}
`).join(""); bindListActions(); } function renderTasks() { if (!state.tareas.length) { document.getElementById("tareas-count-badge").textContent = "0 tareas"; tasksList.innerHTML = emptyState( "No hay tareas registradas", "Agrega la primera tarea para empezar el seguimiento de mantencion." ); return; } const filteredTasks = state.tareas.filter((item) => { const query = normalize(filters.tasks.query); const matchesQuery = !query || normalize([item.title, item.area].join(" ")).includes(query); const matchesStatus = filters.tasks.status === "todos" || item.status === filters.tasks.status; return matchesQuery && matchesStatus; }); document.getElementById("tareas-count-badge").textContent = formatVisibleCount(filteredTasks.length, state.tareas.length, "tareas"); if (!filteredTasks.length) { tasksList.innerHTML = emptyState( "No hay coincidencias para este filtro", "Prueba con otro texto o selecciona un estado distinto." ); return; } tasksList.innerHTML = filteredTasks.map((item) => `

${escapeHtml(item.title)}

${escapeHtml(item.area || "Sin area")} ${escapeHtml(item.date || "Sin fecha")}
${renderBadge(item.status, "task")}
`).join(""); bindListActions(); } function renderIncidents() { if (!state.incidentes.length) { document.getElementById("incidentes-count-badge").textContent = "0 incidentes"; incidentsList.innerHTML = emptyState( "No hay incidentes registrados", "Agrega el primer incidente cuando necesites dejar trazabilidad de un caso." ); return; } const filteredIncidents = state.incidentes.filter((item) => { const query = normalize(filters.incidents.query); const matchesQuery = !query || normalize([item.description, item.source].join(" ")).includes(query); const matchesStatus = filters.incidents.status === "todos" || item.status === filters.incidents.status; const matchesLevel = filters.incidents.level === "todos" || item.level === filters.incidents.level; return matchesQuery && matchesStatus && matchesLevel; }); document.getElementById("incidentes-count-badge").textContent = formatVisibleCount(filteredIncidents.length, state.incidentes.length, "incidentes"); if (!filteredIncidents.length) { incidentsList.innerHTML = emptyState( "No hay coincidencias para este filtro", "Ajusta texto, estado o nivel para volver a ver resultados." ); return; } incidentsList.innerHTML = filteredIncidents.map((item) => `

${escapeHtml(truncate(item.description, 90))}

${escapeHtml(item.source || "Sin origen")} ${escapeHtml(item.date || "Sin fecha")} Nivel ${escapeHtml(item.level)}
${renderBadge(item.status, "incident")}
`).join(""); bindListActions(); } function bindListActions() { document.querySelectorAll("[data-edit]").forEach((button) => { button.onclick = () => startEdit(button.dataset.edit, button.dataset.id); }); document.querySelectorAll("[data-delete]").forEach((button) => { button.onclick = () => { const type = button.dataset.delete; const id = button.dataset.id; if (type === "locale") state.locales = state.locales.filter((item) => item.id !== id); if (type === "task") state.tareas = state.tareas.filter((item) => item.id !== id); if (type === "incident") state.incidentes = state.incidentes.filter((item) => item.id !== id); if (editing[type] === id) resetFormMode(type); saveAndRender(); }; }); document.querySelectorAll("select[data-type]").forEach((select) => { select.onchange = () => { const type = select.dataset.type; const id = select.dataset.id; const item = findItem(type, id); if (!item) return; item.status = select.value; saveAndRender(); }; }); } function startEdit(type, id) { const item = findItem(type, id); if (!item) return; editing[type] = id; if (type === "locale") { const form = document.getElementById("locale-form"); form.number.value = item.number || ""; form.name.value = item.name || ""; form.tenant.value = item.tenant || ""; form.phone.value = item.phone || ""; form.category.value = item.category || ""; form.status.value = item.status || "disponible"; applyFormMode("locale", "Editar local", "Actualizar local", "Editando un local ya registrado."); switchPanel("locales"); document.getElementById("locale-number").focus(); } if (type === "task") { const form = document.getElementById("task-form"); form.title.value = item.title || ""; form.area.value = item.area || ""; form.date.value = item.date || ""; form.status.value = item.status || "pendiente"; applyFormMode("task", "Editar tarea", "Actualizar tarea", "Editando una tarea existente."); switchPanel("tareas"); document.getElementById("task-title").focus(); } if (type === "incident") { const form = document.getElementById("incident-form"); form.description.value = item.description || ""; form.level.value = item.level || "bajo"; form.status.value = item.status || "abierto"; form.date.value = item.date || ""; form.source.value = item.source || ""; applyFormMode("incident", "Editar incidente", "Actualizar incidente", "Editando un incidente existente."); switchPanel("incidentes"); document.getElementById("incident-description").focus(); } } function resetFormMode(type) { editing[type] = null; if (type === "locale") { const form = document.getElementById("locale-form"); form.reset(); form.status.value = "disponible"; applyDefaultFormMode("locale", "Agregar local", "Guardar local"); } if (type === "task") { const form = document.getElementById("task-form"); form.reset(); form.status.value = "pendiente"; applyDefaultFormMode("task", "Nueva tarea", "Guardar tarea"); } if (type === "incident") { const form = document.getElementById("incident-form"); form.reset(); form.level.value = "bajo"; form.status.value = "abierto"; applyDefaultFormMode("incident", "Nuevo incidente", "Guardar incidente"); } } function resetAllFormModes() { resetFormMode("locale"); resetFormMode("task"); resetFormMode("incident"); } function resetAllFilters() { filters.locales.query = ""; filters.locales.status = "todos"; filters.tasks.query = ""; filters.tasks.status = "todos"; filters.incidents.query = ""; filters.incidents.status = "todos"; filters.incidents.level = "todos"; document.getElementById("locales-search").value = ""; document.getElementById("locales-filter-status").value = "todos"; document.getElementById("tasks-search").value = ""; document.getElementById("tasks-filter-status").value = "todos"; document.getElementById("incidents-search").value = ""; document.getElementById("incidents-filter-status").value = "todos"; document.getElementById("incidents-filter-level").value = "todos"; } function applyFormMode(type, title, submitLabel, note) { document.getElementById(`${type}-form-title`).textContent = title; document.getElementById(`${type}-submit`).textContent = submitLabel; document.getElementById(`${type}-cancel`).hidden = false; document.getElementById(`${type}-form-note`).textContent = note; document.getElementById(`${type}-form-note`).hidden = false; } function applyDefaultFormMode(type, title, submitLabel) { document.getElementById(`${type}-form-title`).textContent = title; document.getElementById(`${type}-submit`).textContent = submitLabel; document.getElementById(`${type}-cancel`).hidden = true; document.getElementById(`${type}-form-note`).hidden = true; } function findItem(type, id) { if (type === "locale") return state.locales.find((item) => item.id === id); if (type === "task") return state.tareas.find((item) => item.id === id); if (type === "incident") return state.incidentes.find((item) => item.id === id); return null; } function renderBadge(status, kind) { const localeMap = { disponible: ["badge-blue", "Disponible"], ocupado: ["badge-green", "Ocupado"], reservado: ["badge-yellow", "Reservado"] }; const taskMap = { pendiente: ["badge-yellow", "Pendiente"], "en progreso": ["badge-blue", "En progreso"], completada: ["badge-green", "Completada"] }; const incidentMap = { abierto: ["badge-red", "Abierto"], "en revision": ["badge-yellow", "En revision"], resuelto: ["badge-green", "Resuelto"] }; const maps = { locale: localeMap, task: taskMap, incident: incidentMap }; const [styleClass, label] = maps[kind][status] || ["badge-blue", capitalize(status)]; return `${escapeHtml(label)}`; } function formatVisibleCount(visible, total, label) { return visible === total ? `${total} ${label}` : `${visible} de ${total} ${label}`; } function normalize(value) { return String(value || "") .toLowerCase() .normalize("NFD") .replace(/[\u0300-\u036f]/g, ""); } function emptyState(title, copy) { return `
${escapeHtml(title)}

${escapeHtml(copy)}

`; } function escapeHtml(value) { return String(value) .replaceAll("&", "&") .replaceAll("<", "<") .replaceAll(">", ">") .replaceAll("\"", """) .replaceAll("'", "'"); } function capitalize(value) { if (!value) return ""; return value.charAt(0).toUpperCase() + value.slice(1); } function truncate(value, length) { if (!value) return ""; return value.length > length ? `${value.slice(0, length - 3)}...` : value; } resetAllFormModes(); resetAllFilters(); render();
